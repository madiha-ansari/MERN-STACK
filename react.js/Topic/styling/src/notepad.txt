import React, { useState } from "react";

const App = () => {
  const [bgColor, setBgColor] = useState("green");

  return (
    <div>
      <button
        style={{
          backgroundColor: bgColor,
          color: "white",
          padding: "10px 20px",
          border: "none",
          borderRadius: "5px",
          cursor: "pointer",
        }}
        onClick={() => setBgColor(bgColor === "green" ? "red" : "green")}
      >
        Change Color
      </button>
    </div>
  );
};

export default App;
